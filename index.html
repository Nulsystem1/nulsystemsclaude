# NTV dispatch authorization — compiled from policy rule authorize_dispatch(request): if not request.operator_cert_current: return BLOCK("operator certification expired") if not request.route in approved_geofence: return BLOCK("route outside approved geofence") if request.visibility < policy.min_visibility: return REVIEW("low visibility — human decision") if request.maint_flag_open: return BLOCK("open maintenance flag") return ALLOW(audit=True)
